\newtheorem*{sublemma*}{Sublemma}

\newtheorem{claim}[theorem]{Claim}
\newtheorem*{claim*}{Claim}

\newtheorem{subclaim}[theorem]{Subclaim}
\newtheorem*{subclaim*}{Subclaim}
