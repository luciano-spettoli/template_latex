%\pagestyle{empty}

\ifpdftex
  \usepackage{lmodern}
  \usepackage[T1]{fontenc}
\fi

\ifluatex
  \usepackage{fontspec}
  \usepackage[olddefault, varnothing]{fontsetup}  % New Computer Modern in regular weight

\usepackage[stretch=16]{microtype}
\usepackage[english]{babel}

% Disable hyphenation
%\pretolerance    = 100               % default: 100

  top    = 32mm,
  bottom = 32mm,
  left   = 28mm,
  right  = 28mm
]{geometry}

\usepackage{setspace}
%\onehalfspacing
\setstretch{1.4}

% Disable indentation at beginning of paragraphs; and customise vertical spacing
\usepackage[
  skip    = 0.36\baselineskip plus 0.12\baselineskip minus 0.08\baselineskip,  % vertical spacing between paragraphs;
  tocskip = 0.2\baselineskip  plus 0.06\baselineskip minus 0.04\baselineskip   % vertical spacing between toc entries
]{parskip}

% Disable paragraphs breaking across a pagebreak with only one line on one of the pages,
% and discourage breaking with only a few lines on one side
